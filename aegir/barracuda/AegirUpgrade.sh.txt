_INSTALLER_VERSION=EDIT_INSTALLER_VERSION
_BOA_REPO_NAME=EDIT_BOA_REPO_NAME
_BOA_REPO_GIT_URL=EDIT_BOA_REPO_GIT_URL
_DOMAIN=EDIT_DOMAIN
_AEGIR_VERSION=EDIT_AEGIR_VERSION
_DRUSH_VERSION=EDIT_DRUSH_VERSION
_DEBUG_MODE=EDIT_DEBUG_MODE
_THIS_DB_HOST=EDIT_THIS_DB_HOST

###---### Functions
#
# noticeable messages
msg () {
  echo "Barracuda ==> $*"
}
#
# stop on error
# set -e
export DEBIAN_FRONTEND=noninteractive
#
hostmaster_dr_up () {
  cd $_ROOT
  mv -f drush $_ROOT/backups/drush-pre-$_DISTRO-$_NOW &> /dev/null
  wget -q -U iCab http://ftp.drupal.org/files/projects/drush-$_DRUSH_VERSION.tar.gz
  tar -xzf drush-$_DRUSH_VERSION.tar.gz
  rm -f drush-$_DRUSH_VERSION.tar.gz
  if $_DRUSHCMD help > /dev/null ; then
    msg "`date` UPGRADE INFO: Drush seems to be functioning properly"
  else
    msg "`date` UPGRADE INFO: Drush is broken ($DRUSH help failed)"
    exit 1
  fi
}
#
hostmaster_mv_up () {
  mv -f $_ROOT/.drush/provision $_ROOT/backups/provision-pre-$_DISTRO-$_NOW &> /dev/null
  mv -f $_ROOT/.drush/drush_make $_ROOT/backups/drush_make-pre-$_DISTRO-$_NOW &> /dev/null
  mv -f $_ROOT/.drush/provision_boost $_ROOT/backups/provision_boost-pre-$_DISTRO-$_NOW &> /dev/null
}
#
hostmaster_go_up () {
  if $_DRUSHCMD help | grep "^ provision-install" > /dev/null ; then
    msg "`date` UPGRADE INFO: Provision already seems to be installed"
  else
    msg "`date` UPGRADE INFO: Installing provision backend in $_ROOT/.drush"
    mkdir -p $_ROOT/.drush
    cd $_ROOT/.drush
    if [ "$_AEGIR_VERSION" = "HEAD" ]; then
      git clone $_BOA_REPO_GIT_URL/provision.git &> /dev/null
    else
      wget -q -U iCab http://files.aegirproject.org/provision-$_AEGIR_VERSION.tgz
      tar -xzf provision-$_AEGIR_VERSION.tgz
      rm provision-$_AEGIR_VERSION.tgz
    fi
    wget -q -U iCab http://ftp.drupal.org/files/projects/drush_make-6.x-2.0-beta9.tar.gz
    tar -xzf drush_make-6.x-2.0-beta9.tar.gz
    rm -f drush_make-6.x-2.0-beta9.tar.gz
  fi
  cd $_ROOT/.drush
  git clone git://git.koumbit.net/drupal/contrib/modules/provision_boost &> /dev/null
  cd $_PREV_AEGIR_ROOT
}
#
hostmaster_mi_up () {
  export DEBIAN_FRONTEND=noninteractive
  if [ "$_DEBUG_MODE" = "YES" ] ; then
    $_DRUSHCMD hostmaster-migrate $_DOMAIN $_AEGIR_ROOT -y -d
  else
    $_DRUSHCMD hostmaster-migrate $_DOMAIN $_AEGIR_ROOT -y &> /dev/null
  fi
  export DEBIAN_FRONTEND=text
}

###--------------------###

_NOW=`date +%y%m%d-%H%M`
_LASTNUM=001
_DISTRO=001
_DRUSHCMD="php /var/aegir/drush/drush.php"
_ROOT="/var/aegir"
_AEGIR_DIRN=`ls -l $_ROOT | grep hostmaster | cut -d: -f2 | awk '{ print $2}'`
_PREV_AEGIR_ROOT="$_ROOT/$_AEGIR_DIRN"

      if [ -e "$_ROOT/.drush/hostmaster.alias.drushrc.php" ] ; then
        _THIS_HOSTMASTER=`cat $_ROOT/.drush/hostmaster.alias.drushrc.php | grep "'root'" | cut -d: -f2 | awk '{ print $3}'`
        _THIS_HOSTMASTER=`echo $_THIS_HOSTMASTER | sed "s/[\,']//g"`
      fi

  if [ -e "/var/aegir/.drush/hostmaster.alias.drushrc.php" ] ; then
    _THIS_FRONT=`cat /var/aegir/.drush/hostmaster.alias.drushrc.php | grep "uri'" | cut -d: -f2 | awk '{ print $3}'`
    _THIS_FRONT=`echo $_THIS_FRONT | sed "s/[\,']//g"`


if [ -d "$_ROOT/host_master/001" ] ; then
  if [ ! -d "$_ROOT/host_master/000" ] ; then
    mkdir -p $_ROOT/host_master/000
    touch $_ROOT/host_master/000/placeholder_dont_remove.txt
  fi
fi
if [ -d "$_ROOT/host_master/000" ] ; then
  cd $_ROOT/host_master
  list=([0-9]*)
  last=${list[@]: -1}
  _LASTNUM=$last
  nextnum=00$((10#${last%%[^0-9]*} + 1))
  nextnum=${nextnum: -3}
  _PREV_AEGIR_ROOT_TEST="$_ROOT/host_master/$_LASTNUM"
  if [ -e "$_PREV_AEGIR_ROOT_TEST/sites/$_DOMAIN/settings.php" ] ; then
    msg "`date` UPGRADE INFO: Testing previous install.. OK"
    _DISTRO=${nextnum}
    _PREV_AEGIR_ROOT="$_ROOT/host_master/$_LASTNUM"
  else
    _DEBUG_MODE=YES
    msg "`date` UPGRADE INFO: Testing previous install.. OPS, zombie found, moving it to backups.."
    mv -f $_PREV_AEGIR_ROOT_TEST $_ROOT/backups/hostmaster-pre-$_DISTRO-$_NOW &> /dev/null
    cd $_ROOT/host_master
    list=([0-9]*)
    last=${list[@]: -1}
    _LASTNUM=$last
    nextnum=00$((10#${last%%[^0-9]*} + 1))
    nextnum=${nextnum: -3}
    _DISTRO=${nextnum}
    _PREV_AEGIR_ROOT_TEST="$_ROOT/host_master/$_LASTNUM"
    if [ -e "$_PREV_AEGIR_ROOT_TEST/sites/$_DOMAIN/settings.php" ] ; then
      msg "`date` UPGRADE INFO: Testing previous install.. OK"
      _DISTRO=${nextnum}
      _PREV_AEGIR_ROOT="$_ROOT/host_master/$_LASTNUM"
    fi
  fi
fi
_AEGIR_ROOT="$_ROOT/host_master/$_DISTRO"
if [ -d "$_AEGIR_ROOT" ] ; then
  msg "`date` UPGRADE INFO: FATAL ERROR: $_AEGIR_ROOT already exist! Exiting.."
  exit 1
fi
mkdir -p $_ROOT/host_master
chmod 711 $_ROOT/host_master &> /dev/null
msg "`date` UPGRADE INFO: Simple check if previous Hostmaster version was a valid install"
if [ -e "$_PREV_AEGIR_ROOT/sites/$_DOMAIN/settings.php" ] ; then
  msg "`date` UPGRADE INFO: Test OK, we can proceed with the Hostmaster upgrade"
  hostmaster_dr_up
  hostmaster_mv_up
  hostmaster_go_up
  hostmaster_mi_up
elif [ ! -d "/var/aegir/.drush/provision/http" ] ; then
  msg "`date` UPGRADE ERROR: Something (1) is wrong, we should stop now.."
  msg "`date` UPGRADE ERROR: Please report this in the GitHub queue"
  msg "BYE!"
  exit 1
else
  msg "`date` UPGRADE ERROR: Something (2) is wrong, we should stop now.."
  msg "`date` UPGRADE ERROR: Please report this in the GitHub queue"
  msg "BYE!"
  exit 1
fi
