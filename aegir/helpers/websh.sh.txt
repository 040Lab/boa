#!/bin/bash

PATH=/usr/local/bin:/usr/bin:/bin
SHELL=/bin/bash
_ALL="$@"
_LTD_GID=$(id -nG $USER)

if [[ "$_LTD_GID" =~ "lshellg"($) ]] ; then
  if [ "$1" = "-c" ] ; then
    _TARGET_CMD="$2 $3 $4 $5"
    if [[ "$2" =~ (^)"/usr/" ]] || [[ "$2" =~ (^)"/bin/" ]] ; then
      _IN_PATH=YES
    else
      _WHICH_TEST=$(which $2)
      if [[ "$_WHICH_TEST" =~ (^)"/usr/" ]] || [[ "$_WHICH_TEST" =~ (^)"/bin/" ]] ; then
        _IN_PATH=YES
      else
        _IN_PATH=NO
      fi
    fi
  else
    _TARGET_CMD="$@"
    if [[ "$1" =~ (^)"/usr/" ]] || [[ "$1" =~ (^)"/bin/" ]] ; then
      _IN_PATH=YES
    else
      _WHICH_TEST=$(which $1)
      if [[ "$_WHICH_TEST" =~ (^)"/usr/" ]] || [[ "$_WHICH_TEST" =~ (^)"/bin/" ]] ; then
        _IN_PATH=YES
      else
        _IN_PATH=NO
      fi
    fi
  fi
  if [ "$_IN_PATH" = "YES" ] ; then
    exec $_TARGET_CMD
    exit 0
  else
    exit 1
  fi
else
  if [ "$1" = "-c" ] ; then
    _TARGET_CMD="$2 $3 $4 $5"
    exec $_TARGET_CMD
    exit 0
  else
    _TARGET_CMD="$@"
    exec $_TARGET_CMD
    exit 0
  fi
fi
